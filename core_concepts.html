core concepts

  • select & selectAll 
    var svg = d3.select("svg");
var circles = svg.selectAll("circle");
  • bind data 
    circles.data([32, 57, 112]);
  • enter & exit 
    circles.enter().append("circle");
circles.exit().remove();
  • dynamic properties 
    circles.attr("r", function(datum, index) {
    return datum * 2;
});
